```bash
FILTER_BRANCH_SQUELCH_WARNING=1 git filter-branch -f --env-filter '

export GIT_COMMITTER_EMAIL="joesubsho@gmail.com"
export GIT_AUTHOR_NAME="Joe Ho"

git stash pop  # Restore working changes
```
